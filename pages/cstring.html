void strcopy(char *a, char *b)
{
    for (int i = 0; a[i] != '\0'; i++)
    {
        b[i] = a[i];
        if (a[i + 1] == '\0')
            b[i + 1] = '\0'; //specifying end of string to avoid junk value here
    }
}

int strlength(char *strl)
{
    int i = 0;
    while (strl[i] != 0)
    {
        i++;
    }
    return i;
}

void strconcat(char *p, char *q, char *res)
{
    while (*p)
    {
        *res++ = *p++;
    }
    while (*q)
    {
        *res++ = *q++;
    }
    *res = '\0'; //specifying end of string to avoid junk value here
}

int strcompare(char *x, char *y)
{
    for (int i = 0; x[i] != '\0'; i++)
    {
        if (y[i] != x[i])
        {
            return 0;
        }
    }
    return 1;
}

void strreverse(char *s)
{
    int i = 0, j = 0;
    // printf("\n String: %s", s);
    // printf("\n Char 1: %c", s[0]);

    while (s[j] != 0)
    {
        j++;
    }
    // printf("\n test size : %d", j);
    j--; //we dont want '\0' as first character in string
    char temp;
    while (i < j)
    {
        temp = s[i];
        s[i] = s[j];
        s[j] = temp;
        i++;
        j--;
    }
}

void strlower(char *stlwr)
{
    while (*stlwr)
    {
        if (*stlwr <= 'Z' && *stlwr >= 'A')
            *stlwr += 32;
        *stlwr++;
    }
}

void strupper(char *stupr)
{
    while (*stupr)
    {
        if (*stupr <= 'z' && *stupr >= 'a')
            *stupr -= 32;
        *stupr++;
    }
}
